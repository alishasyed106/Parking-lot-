📄 ParkingLot.java — Observer Subject 
// Subject (Observable) — notifies observers on slot state changes
@Component
public class ParkingLot implements SlotSubject {

    private final List<ParkingSlot> slots;
    private final List<SlotObserver> observers = new ArrayList<>();
    private AllocationStrategy strategy;

    public ParkingLot(List<ParkingSlot> slots,
                       AllocationStrategy strategy) {
        this.slots = slots;
        this.strategy = strategy;
    }

    @Override
    public void registerObserver(SlotObserver observer) {
        observers.add(observer);
    }

    @Override
    public void notifyObservers(SlotEvent event) {
        observers.forEach(o -> o.update(event));
    }

    public Optional<ParkingSlot> parkVehicle(Vehicle vehicle) {
        return strategy
            .allocate(slots, vehicle)
            .map(slot -> {
                slot.assign(vehicle);
                notifyObservers(SlotEvent.occupied(slot, vehicle));
                return slot;
            });
    }

    public void releaseSlot(String slotId) {
        slots.stream()
             .filter(s -> s.getId().equals(slotId))
             .findFirst()
             .ifPresent(slot -> {
                 slot.release();
                 notifyObservers(SlotEvent.available(slot));
             });
    }
}
// Open/Closed Principle — add strategies without modifying core
public interface AllocationStrategy {
    Optional<ParkingSlot> allocate(List<ParkingSlot> slots,
                                      Vehicle vehicle);
}

// Circular Strategy — wraps around, evenly distributes wear
@Component
public class CircularAllocationStrategy implements AllocationStrategy {

    private int pointer = 0;

    @Override
    public Optional<ParkingSlot> allocate(List<ParkingSlot> slots,
                                              Vehicle vehicle) {
        int n = slots.size(), checked = 0;
        while (checked++ < n) {
            ParkingSlot slot = slots.get(pointer % n);
            pointer = (pointer + 1) % n;
            if (slot.isAvailable()) return Optional.of(slot);
        }
        return Optional.empty();
    }
}

// Buffer Strategy — reserves N slots as emergency buffer
@Component
public class BufferAllocationStrategy implements AllocationStrategy {

    private final int bufferSize;

    public BufferAllocationStrategy(@Value("${parking.buffer.size:4}") int bufferSize) {
        this.bufferSize = bufferSize;
    }

    @Override
    public Optional<ParkingSlot> allocate(List<ParkingSlot> slots,
                                              Vehicle vehicle) {
        List<ParkingSlot> available = slots.stream()
            .filter(ParkingSlot::isAvailable)
            .filter(s -> !s.isBuffer())
            .toList();

        if (available.size() <= bufferSize && !vehicle.isEmergency())
            return Optional.empty();

        return available.stream().findFirst();
    }
}
// notifier.go — concurrent notification dispatcher via goroutines
package notifier

import (
    "context"
    "log/slog"
    "sync"
    "time"
)

type SlotEvent struct {
    SlotID    string
    Status    string
    VehicleID string
    Timestamp time.Time
}

type Observer interface {
    Notify(ctx context.Context, event SlotEvent) error
}

type NotifierHub struct {
    mu        sync.RWMutex
    observers []Observer
    events    chan SlotEvent
}

func NewNotifierHub() *NotifierHub {
    h := &NotifierHub{events: make(chan SlotEvent, 256)}
    go h.dispatch()
    return h
}

func (h *NotifierHub) dispatch() {
    for event := range h.events {
        h.mu.RLock()
        obs := h.observers
        h.mu.RUnlock()

        var wg sync.WaitGroup
        for _, o := range obs {
            wg.Add(1)
            go func(observer Observer) {
                defer wg.Done()
                ctx, cancel := context.WithTimeout(
                    context.Background(), 5*time.Second)
                defer cancel()
                if err := observer.Notify(ctx, event); err != nil {
                    slog.Error("observer failed", "err", err)
                }
            }(o)
        }
        wg.Wait()
    }
}
// RESTful controller — Spring Boot 3, SOLID-compliant
@RestController
@RequestMapping("/api/v1/parking")
@RequiredArgsConstructor
@Slf4j
public class ParkingController {

    private final ParkingService parkingService;

    @PostMapping("/park")
    @ResponseStatus(HttpStatus.CREATED)
    public ResponseEntity<ParkingResponse> parkVehicle(
            @RequestBody @Valid ParkingRequest request) {

        return parkingService
            .parkVehicle(request)
            .map(slot -> ResponseEntity
                .status(HttpStatus.CREATED)
                .body(ParkingResponse.from(slot)))
            .orElseThrow(() -> new LotFullException(
                "No available slots matching strategy"));
    }

    @DeleteMapping("/release/{slotId}")
    public ResponseEntity<Void> releaseSlot(
            @PathVariable String slotId) {
        parkingService.releaseSlot(slotId);
        return ResponseEntity.noContent().build();
    }

    @GetMapping("/status")
    public LotStatusResponse getLotStatus() {
        return parkingService.getLotStatus();
    }

    @GetMapping("/slots")
    public Page<SlotDTO> getSlots(Pageable pageable,
            @RequestParam(required = false) SlotStatus status) {
        return parkingService.findSlots(status, pageable);
    }
}

CREATE TABLE parking_lots (
    id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    name        VARCHAR(100) NOT NULL,
    strategy    VARCHAR(20)  NOT NULL CHECK (strategy IN ('CIRCULAR', 'BUFFER')),
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE parking_slots (
    id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    lot_id      UUID        NOT NULL REFERENCES parking_lots(id),
    slot_code   VARCHAR(10)  NOT NULL,
    floor       SMALLINT    NOT NULL DEFAULT 1,
    status      VARCHAR(20)  NOT NULL DEFAULT 'AVAILABLE',
    is_buffer   BOOLEAN     NOT NULL DEFAULT FALSE,
    version     BIGINT      NOT NULL DEFAULT 0,  -- optimistic lock
    UNIQUE (lot_id, slot_code)
);

CREATE TABLE bookings (
    id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    slot_id     UUID        NOT NULL REFERENCES parking_slots(id),
    vehicle_id  VARCHAR(30)  NOT NULL,
    is_emergency BOOLEAN    NOT NULL DEFAULT FALSE,
    parked_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
    released_at TIMESTAMPTZ
);

CREATE INDEX idx_slots_lot_status ON parking_slots (lot_id, status);
CREATE INDEX idx_bookings_vehicle ON bookings (vehicle_id);
CREATE INDEX idx_bookings_active  ON bookings (slot_id) WHERE released_at IS NULL;
Features

What's Under the Hood

⟳
Circular Allocation
Slot assignment wraps around the lot using a circular pointer, evenly distributing vehicle throughput and minimizing wear concentration on any single zone.
⊞
Buffer Allocation
Configurable emergency buffer reserves N slots exclusively for high-priority vehicles. Regular traffic is blocked from buffer slots when occupancy is critical.
👁️
Real-Time Observer Notifications
Dashboard, SMS operator, mobile app, and billing service all subscribe independently. Adding new observers requires zero changes to the ParkingLot core.
🧱
SOLID Architecture
Single Responsibility on every class. Open/Closed for new strategies. Liskov-safe slot types. Interface Segregation on observer roles. Dependency Inversion throughout.
⚡
Concurrent Notification Dispatch
The Golang service dispatches notifications to all observers concurrently via goroutines with per-observer timeout contexts, ensuring one slow observer never blocks others.
🔒
Optimistic Locking
Version columns in PostgreSQL prevent double-allocation races under concurrent load. Conflicting writes trigger retries via Spring's @Retryable, not slot corruption.
